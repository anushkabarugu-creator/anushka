public class Inheritance{
    public static void main(String[] args){
        Hacker h=new Hacker();
        h.viewDetails();
        h.changeDetails();
        System.out.println("change values");
        h.viewDetails();
    }
}
class Credit{
    int cardNo=111234;
    int pinNo=503;
}
class Hacker extends Credit{
    void viewDetails()
    {
        System.out.println(cardNo);
        System.out.println(pinNo);
    }
    void changeDetails()
    {
        cardNo=111111;
        pinNo=563;
    }
}
